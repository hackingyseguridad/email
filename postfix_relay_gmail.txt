### Notas para la configuracion SMTP Postfix en localhost y envio de email desde consola Kali Linux
### Utilizar como relay el servidor de correo saliente de Google SMTP GMAIL

### /etc/postfix/main.cf  ejemplo de fichero main.cf para envio con gmail

smtpd_banner = $myhostname ESMTP
biff = no
append_dot_mydomain = no
inet_interfaces = loopback-only
myhostname = localhost
mydomain = localdomain
myorigin = $mydomain
mydestination = $myhostname, localhost
mynetworks = 127.0.0.0/8 [::1]/128
#smtp_tls_CAfile = /etc/ssl/certs/MyRootCA.crt
#smtp_use_tls = yes
relayhost = [smtp.gmail.com]:587            # - Usa Gmail como relay
smtp_use_tls = yes                          # - Habilita TLS
smtp_sasl_auth_enable = yes                  # - Habilita autenticación
smtp_sasl_password_maps = hash:/etc/postfix/sasl_passwd      # ← Credenciales
smtp_sasl_security_options = noanonymous                   # ← Sin anónimos
